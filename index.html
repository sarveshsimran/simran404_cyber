import requests
import ssl
import socket
from urllib.parse import urlparse
import json

class SecurityScanner:
    def __init__(self, target_url):
        self.target_url = target_url
        self.results = {}
    
    def check_ssl_certificate(self):
        try:
            parsed = urlparse(self.target_url)
            context = ssl.create_default_context()
            with socket.create_connection((parsed.netloc, 443), timeout=10) as sock:
                with context.wrap_socket(sock, server_hostname=parsed.netloc) as ssock:
                    cert = ssock.getpeercert()
                    self.results['ssl_valid'] = True
                    self.results['ssl_expiry'] = cert.get('notAfter')
        except Exception as e:
            self.results['ssl_valid'] = False
            self.results['ssl_error'] = str(e)
    
    def check_headers(self):
        try:
            response = requests.get(self.target_url, timeout=10)
            headers = response.headers
            
            security_headers = {
                'X-Content-Type-Options': 'nosniff' in headers.get('X-Content-Type-Options', ''),
                'X-Frame-Options': 'DENY' in headers.get('X-Frame-Options', '') or 'SAMEORIGIN' in headers.get('X-Frame-Options', ''),
                'Strict-Transport-Security': 'max-age' in headers.get('Strict-Transport-Security', ''),
                'Content-Security-Policy': 'default-src' in headers.get('Content-Security-Policy', '')
            }
            
            self.results['security_headers'] = security_headers
        except Exception as e:
            self.results['headers_error'] = str(e)
    
    def scan(self):
        print(f"Starting security scan for: {self.target_url}")
        self.check_ssl_certificate()
        self.check_headers()
        return self.results

# Usage example
if __name__ == "__main__":
    scanner = SecurityScanner("https://example.com")
    results = scanner.scan()
    print(json.dumps(results, indent=2))
